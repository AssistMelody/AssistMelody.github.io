export class UnlessDirective{
  constructor(
    private view:ViewContainerRef,
    private temp:TemplateRef<any>
  ){}
  @Input() set appUnless(val:Boolean){
    if (!val) {
      this.view.createEmbeddedView(this.temp);

      this.view.clear()
    }
  };
}
